#include <esp_now.h>
#include <WiFi.h>
#include <esp_wifi.h>

// ========== 可调参数 ==========
uint32_t g_activeDelayUs = 1000;   // 活动轮询微秒
uint32_t g_idleDelayMs = 100;
uint32_t g_idleTimeoutMs = 60000;
uint8_t  g_txPower = 20;

// ========== 发射端 MAC 地址（与发射端代码中的接收端MAC相同？注意配对关系）==========
// 实际上接收端需要知道发射端的MAC才能单播发送命令，但也可以广播或多对一。简单起见，我们固定发射端MAC。
// 请修改为实际的发射端MAC地址
uint8_t senderMac[] = {0x24, 0xEC, 0x4A, 0x11, 0x9F, 0x3D};

typedef struct __attribute__((packed)) {
  bool up; bool down; bool left; bool right;
  bool a; bool b; bool x; bool y;
  bool lb; bool rb; bool lt; bool rt;
  bool start; bool select; bool home; bool capture;
  bool ls; bool rs;
} GamepadState;

GamepadState currentState;
GamepadState lastSentState;
bool newData = false;

unsigned long lastActiveTime = 0;
unsigned long currentDelayUs = g_activeDelayUs;

bool isAnyButtonPressed(const GamepadState &state) {
  return state.up    || state.down  || state.left  || state.right ||
         state.a     || state.b     || state.x     || state.y     ||
         state.lb    || state.rb    || state.lt    || state.rt    ||
         state.start || state.select|| state.home  || state.capture ||
         state.ls    || state.rs;
}

void sendSerialFrame(const GamepadState &s) {
  char buf[19];
  buf[0]  = s.up      ? '1' : '0';
  buf[1]  = s.down    ? '1' : '0';
  buf[2]  = s.left    ? '1' : '0';
  buf[3]  = s.right   ? '1' : '0';
  buf[4]  = s.a       ? '1' : '0';
  buf[5]  = s.b       ? '1' : '0';
  buf[6]  = s.x       ? '1' : '0';
  buf[7]  = s.y       ? '1' : '0';
  buf[8]  = s.lb      ? '1' : '0';
  buf[9]  = s.rb      ? '1' : '0';
  buf[10] = s.lt      ? '1' : '0';
  buf[11] = s.rt      ? '1' : '0';
  buf[12] = s.start   ? '1' : '0';
  buf[13] = s.select  ? '1' : '0';
  buf[14] = s.home    ? '1' : '0';
  buf[15] = s.capture ? '1' : '0';
  buf[16] = s.ls      ? '1' : '0';
  buf[17] = s.rs      ? '1' : '0';
  buf[18] = '\n';
  Serial.write(buf, 19);
}

void onDataRecv(const esp_now_recv_info_t *info, const uint8_t *data, int len) {
  if (len == sizeof(GamepadState)) {
    memcpy(&currentState, data, sizeof(GamepadState));
    newData = true;
  } else {
    // 可能是来自发射端的文本回应（状态信息）
    char msg[128];
    if (len < sizeof(msg)-1) {
      memcpy(msg, data, len);
      msg[len] = '\0';
      // 添加前缀以便PC区分
      Serial.print("#");
      Serial.println(msg);
    }
  }
}

void setup() {
  Serial.begin(460800);
  WiFi.mode(WIFI_STA);
  if (esp_now_init() != ESP_OK) {
    Serial.println("ESP-NOW init failed");
    return;
  }
  esp_wifi_set_max_tx_power(g_txPower);
  esp_now_register_recv_cb(onDataRecv);
  // 添加发射端为对等方（以便发送命令）
  esp_now_peer_info_t peerInfo = {};
  memcpy(peerInfo.peer_addr, senderMac, 6);
  peerInfo.channel = 0;
  peerInfo.encrypt = false;
  esp_now_add_peer(&peerInfo);

  memset(&lastSentState, 0, sizeof(lastSentState));
  lastActiveTime = millis();
  Serial.println("READY");   // 按键帧识别线
}

void loop() {
  // 处理从PC串口来的命令（转发给发射端）
  if (Serial.available()) {
    static char cmdBuf[128];
    static int idx = 0;
    while (Serial.available()) {
      char c = Serial.read();
      if (c == '\n' || idx >= sizeof(cmdBuf)-1) {
        cmdBuf[idx] = '\0';
        if (strlen(cmdBuf) > 0) {
          // 转发给发射端
          esp_now_send(senderMac, (uint8_t*)cmdBuf, strlen(cmdBuf));
        }
        idx = 0;
        break;
      } else {
        cmdBuf[idx++] = c;
      }
    }
  }

  if (newData) {
    newData = false;
    if (memcmp(&currentState, &lastSentState, sizeof(GamepadState)) != 0) {
      sendSerialFrame(currentState);
      lastSentState = currentState;
    }
    if (isAnyButtonPressed(currentState)) {
      lastActiveTime = millis();
    }
  }

  // 动态轮询延迟
  if (isAnyButtonPressed(currentState)) {
    currentDelayUs = g_activeDelayUs;
  } else {
    if (millis() - lastActiveTime >= g_idleTimeoutMs) {
      currentDelayUs = g_idleDelayMs * 1000UL;
    } else {
      currentDelayUs = g_activeDelayUs;
    }
  }
  delayMicroseconds(currentDelayUs);
}
