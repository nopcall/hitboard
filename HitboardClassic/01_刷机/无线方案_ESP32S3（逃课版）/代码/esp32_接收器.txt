#include <esp_now.h>
#include <WiFi.h>
#include <esp_wifi.h>          // 添加功率控制头文件

// ==================================================
// 统一回报率参数（1000Hz 活动，10Hz 空闲）
// ==================================================
#define ACTIVE_DELAY_US    1000          // 有按键时 1000us → 1000Hz
#define IDLE_DELAY_MS      100           // 无按键时 100ms → 10Hz
#define IDLE_TIMEOUT_MS    60000         // 无按键持续 60 秒后降频

// 无线发射功率设置（与发射端保持一致，单位：0.25 dBm）
#define TX_POWER           20            // 5 dBm，约 3.16 mW
// ==================================================

typedef struct __attribute__((packed)) {
  bool up; bool down; bool left; bool right;
  bool a; bool b; bool x; bool y;
  bool lb; bool rb; bool lt; bool rt;
  bool start; bool select; bool home; bool capture;
  bool ls; bool rs;
} GamepadState;

GamepadState currentState;
GamepadState lastSentState;
bool newData = false;

unsigned long lastActiveTime = 0;
unsigned long currentDelayUs = ACTIVE_DELAY_US;

bool isAnyButtonPressed(const GamepadState &state) {
  return state.up    || state.down  || state.left  || state.right ||
         state.a     || state.b     || state.x     || state.y     ||
         state.lb    || state.rb    || state.lt    || state.rt    ||
         state.start || state.select|| state.home  || state.capture ||
         state.ls    || state.rs;
}

void sendSerialFrame(const GamepadState &s) {
  char buf[19];
  buf[0]  = s.up      ? '1' : '0';
  buf[1]  = s.down    ? '1' : '0';
  buf[2]  = s.left    ? '1' : '0';
  buf[3]  = s.right   ? '1' : '0';
  buf[4]  = s.a       ? '1' : '0';
  buf[5]  = s.b       ? '1' : '0';
  buf[6]  = s.x       ? '1' : '0';
  buf[7]  = s.y       ? '1' : '0';
  buf[8]  = s.lb      ? '1' : '0';
  buf[9]  = s.rb      ? '1' : '0';
  buf[10] = s.lt      ? '1' : '0';
  buf[11] = s.rt      ? '1' : '0';
  buf[12] = s.start   ? '1' : '0';
  buf[13] = s.select  ? '1' : '0';
  buf[14] = s.home    ? '1' : '0';
  buf[15] = s.capture ? '1' : '0';
  buf[16] = s.ls      ? '1' : '0';
  buf[17] = s.rs      ? '1' : '0';
  buf[18] = '\n';
  Serial.write(buf, 19);
}

void onDataRecv(const esp_now_recv_info_t *info, const uint8_t *data, int len) {
  if (len == sizeof(GamepadState)) {
    memcpy(&currentState, data, sizeof(GamepadState));
    newData = true;
  }
}

void setup() {
  Serial.begin(460800);
  WiFi.mode(WIFI_STA);
  if (esp_now_init() != ESP_OK) {
    Serial.println("ESP-NOW init failed");
    return;
  }
  // 设置发射功率（限制ACK的功率）
  esp_wifi_set_max_tx_power(TX_POWER);
  esp_now_register_recv_cb(onDataRecv);
  memset(&lastSentState, 0, sizeof(lastSentState));
  lastActiveTime = millis();
  Serial.printf("接收端已就绪 (1000Hz, TX_POWER=%d)\n", TX_POWER);
}

void loop() {
  if (newData) {
    newData = false;

    if (memcmp(&currentState, &lastSentState, sizeof(GamepadState)) != 0) {
      sendSerialFrame(currentState);
      lastSentState = currentState;
    }

    if (isAnyButtonPressed(currentState)) {
      lastActiveTime = millis();
    }
  }

  // 动态调整轮询间隔（微秒单位）
  if (isAnyButtonPressed(currentState)) {
    if (currentDelayUs != ACTIVE_DELAY_US) {
      currentDelayUs = ACTIVE_DELAY_US;
    }
  } else {
    if (millis() - lastActiveTime >= IDLE_TIMEOUT_MS) {
      unsigned long idleUs = IDLE_DELAY_MS * 1000UL;
      if (currentDelayUs != idleUs) {
        currentDelayUs = idleUs;
      }
    } else {
      if (currentDelayUs != ACTIVE_DELAY_US) {
        currentDelayUs = ACTIVE_DELAY_US;
      }
    }
  }

  delayMicroseconds(currentDelayUs);
}
