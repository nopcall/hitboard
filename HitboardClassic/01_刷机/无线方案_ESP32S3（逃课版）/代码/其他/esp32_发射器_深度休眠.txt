#include <esp_now.h>
#include <WiFi.h>
#include <esp_wifi.h>
#include <esp_sleep.h>      // 新增：Light-sleep 相关 API

// ==================================================
// 统一回报率参数
// ==================================================
#define ACTIVE_DELAY_US    1000
#define IDLE_DELAY_MS      100
#define IDLE_TIMEOUT_MS    60000
#define TX_POWER           20

// ==================================================
// 按键唤醒引脚配置（请按实际接线修改）
// ==================================================
#define WAKEUP_PIN_SELECT  14    // select 按键连接的 GPIO
#define WAKEUP_PIN_START   15    // start  按键连接的 GPIO

// ==================================================

typedef struct __attribute__((packed)) {
  bool up; bool down; bool left; bool right;
  bool a; bool b; bool x; bool y;
  bool lb; bool rb; bool lt; bool rt;
  bool start; bool select; bool home; bool capture;
  bool ls; bool rs;
} GamepadState;

GamepadState currentState;
GamepadState lastSentState;
bool newData = false;

unsigned long lastActiveTime = 0;
unsigned long currentDelayUs = ACTIVE_DELAY_US;

// 空闲计时：用于判断是否进入 Light-sleep
unsigned long lastSleepAttempt = 0;
const unsigned long SLEEP_CHECK_INTERVAL = 5000; // 每5秒检查一次是否需要睡

// --------------------------------------------------
// 工具函数
// --------------------------------------------------
bool isAnyButtonPressed(const GamepadState &state) {
  return state.up    || state.down  || state.left  || state.right ||
         state.a     || state.b     || state.x     || state.y     ||
         state.lb    || state.rb    || state.lt    || state.rt    ||
         state.start || state.select|| state.home  || state.capture ||
         state.ls    || state.rs;
}

void sendSerialFrame(const GamepadState &s) {
  char buf[19];
  buf[0]  = s.up      ? '1' : '0';
  buf[1]  = s.down    ? '1' : '0';
  buf[2]  = s.left    ? '1' : '0';
  buf[3]  = s.right   ? '1' : '0';
  buf[4]  = s.a       ? '1' : '0';
  buf[5]  = s.b       ? '1' : '0';
  buf[6]  = s.x       ? '1' : '0';
  buf[7]  = s.y       ? '1' : '0';
  buf[8]  = s.lb      ? '1' : '0';
  buf[9]  = s.rb      ? '1' : '0';
  buf[10] = s.lt      ? '1' : '0';
  buf[11] = s.rt      ? '1' : '0';
  buf[12] = s.start   ? '1' : '0';
  buf[13] = s.select  ? '1' : '0';
  buf[14] = s.home    ? '1' : '0';
  buf[15] = s.capture ? '1' : '0';
  buf[16] = s.ls      ? '1' : '0';
  buf[17] = s.rs      ? '1' : '0';
  buf[18] = '\n';
  Serial.write(buf, 19);
}

void onDataRecv(const esp_now_recv_info_t *info, const uint8_t *data, int len) {
  if (len == sizeof(GamepadState)) {
    memcpy(&currentState, data, sizeof(GamepadState));
    newData = true;
  }
}

// --------------------------------------------------
// 唤醒引脚配置（支持两个按键任意一个唤醒）
// --------------------------------------------------
void configureWakeupPins() {
  // 将两个按键引脚配置为输入上拉
  pinMode(WAKEUP_PIN_SELECT, INPUT_PULLUP);
  pinMode(WAKEUP_PIN_START,  INPUT_PULLUP);
  
  // 使能这两个 GPIO 作为唤醒源（低电平唤醒）
  gpio_wakeup_enable((gpio_num_t)WAKEUP_PIN_SELECT, GPIO_INTR_LOW_LEVEL);
  gpio_wakeup_enable((gpio_num_t)WAKEUP_PIN_START,  GPIO_INTR_LOW_LEVEL);
  
  // 启用 GPIO 唤醒功能（使用 ext1 模式支持多个引脚）
  esp_sleep_enable_gpio_wakeup();
}

// --------------------------------------------------
// 进入 Light-sleep 前停止 Wi-Fi，唤醒后重启
// --------------------------------------------------
void goToLightSleep() {
  Serial.println("进入 Light-sleep，等待 select 或 start 按键唤醒...");
  Serial.flush();
  
  // 1. 停止 Wi-Fi 和 ESP-NOW 以大幅降功耗
  esp_now_deinit();      // 可选，也可以只停 wifi
  esp_wifi_stop();
  
  // 2. 配置唤醒源（两个按键）
  configureWakeupPins();
  
  // 3. （可选）设置一个超时唤醒，比如每隔 10 秒自动醒一次检查状态，防止死睡
  // esp_sleep_enable_timer_wakeup(10 * 1000000);
  
  // 4. 进入 Light-sleep
  esp_light_sleep_start();
  
  // 5. 唤醒后恢复 Wi-Fi 和 ESP-NOW
  Serial.println("唤醒！重新初始化 Wi-Fi / ESP-NOW");
  WiFi.mode(WIFI_STA);
  esp_wifi_start();
  if (esp_now_init() != ESP_OK) {
    Serial.println("ESP-NOW 重连失败");
    return;
  }
  esp_now_register_recv_cb(onDataRecv);
  esp_wifi_set_max_tx_power(TX_POWER);
  
  // 重置活动计时，让设备回到 ACTIVE 模式
  lastActiveTime = millis();
  currentDelayUs = ACTIVE_DELAY_US;
}

// --------------------------------------------------
// setup
// --------------------------------------------------
void setup() {
  Serial.begin(460800);
  WiFi.mode(WIFI_STA);
  esp_wifi_set_max_tx_power(TX_POWER);
  
  if (esp_now_init() != ESP_OK) {
    Serial.println("ESP-NOW init failed");
    return;
  }
  esp_now_register_recv_cb(onDataRecv);
  
  memset(&lastSentState, 0, sizeof(lastSentState));
  lastActiveTime = millis();
  lastSleepAttempt = millis();
  
  Serial.printf("接收端已就绪 (1000Hz, TX_POWER=%d)\n", TX_POWER);
}

// --------------------------------------------------
// loop
// --------------------------------------------------
void loop() {
  // 1. 处理 ESP-NOW 接收的数据
  if (newData) {
    newData = false;
    if (memcmp(&currentState, &lastSentState, sizeof(GamepadState)) != 0) {
      sendSerialFrame(currentState);
      lastSentState = currentState;
    }
    if (isAnyButtonPressed(currentState)) {
      lastActiveTime = millis();
    }
  }
  
  // 2. 动态调整延迟（1000Hz 或 10Hz）
  if (isAnyButtonPressed(currentState)) {
    currentDelayUs = ACTIVE_DELAY_US;
  } else {
    if (millis() - lastActiveTime >= IDLE_TIMEOUT_MS) {
      currentDelayUs = IDLE_DELAY_MS * 1000UL;
    } else {
      currentDelayUs = ACTIVE_DELAY_US;
    }
  }
  
  // 3. 如果当前处于空闲模式（10Hz），并且已经空闲足够久，尝试进入 Light-sleep
  //    注意：只有完全无按键时才考虑睡眠，且避免频繁进睡唤醒
  if (!isAnyButtonPressed(currentState) && (millis() - lastActiveTime >= IDLE_TIMEOUT_MS)) {
    // 为避免每 100ms 都检查一次，增加一个间隔判断
    if (millis() - lastSleepAttempt >= SLEEP_CHECK_INTERVAL) {
      lastSleepAttempt = millis();
      goToLightSleep();   // 进入 Light-sleep，唤醒后函数才返回
    }
  }
  
  // 4. 等待（使用 delayMicroseconds 或更短延时）
  if (currentDelayUs < 4000) {
    delayMicroseconds(currentDelayUs);
  } else {
    delay(currentDelayUs / 1000);  // 空闲时用 delay 可释放 CPU 时间片
  }
}
